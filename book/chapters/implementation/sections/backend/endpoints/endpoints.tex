\subsubsection{\lstinline[language=http]{POST /register}}
Rejestracja nowego użytkownika. \\
\textbf{Request body:}
\begin{lstlisting}[language=json]
{
  "email": "string",
  "password": "string"
}
\end{lstlisting}
\textbf{Responses:}
\begin{itemize}
    \item 200: Rejestracja powiodła się.
    \item 400: Błąd walidacji danych.
\begin{lstlisting}[language=json]
{
  "type": "string",
  "title": "string",
  "status": 0,
  "detail": "string",
  "instance": "string",
  "errors": {
    "additionalProp1": [
      "string"
    ],
    "additionalProp2": [
      "string"
    ],
    "additionalProp3": [
      "string"
    ]
  }
}
\end{lstlisting}
\end{itemize}
