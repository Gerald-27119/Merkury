\subsection{Funkcjonalności dla logowania i rejestracji}
\label{subsec:funkcjonalnosci-dla-logowania i rejestracji}

\begin{requirementstab}[placement=H, label={tab:requirements:func1},caption={}]
    \id{jednoznaczny symbol np. FO1, FO2 .. }
    \priority{M}
    \name{Logowanie i rejestracja}
    \descr{Jako użytkownik chcę mieć możliwość zalogowania się do aplikacji
    korzystając z formularza lub poprzez użycie konta w Google lub GitHub.}
    \acceptcrit{Użytkownik może zalogować się do aplikacji zarówno poprzez
    użycie standardowego formularza jak i przez użycie swojego konta w
    serwisie Google lub GitHub.}
    \inputdata{Dane użytkownika: email, hasło, nazwa użytkownika.}
    \preconditions{Użytkownik nie zalogowany.}
    \postconditions{Działające formularze rejestracji i logowania oraz
    możliwość zalogowania za pomocą konta Googla i Githuba.}
    \exceptions{}
    \implementation{}
    \sholder{}
    \reqrelated{}
\end{requirementstab}
