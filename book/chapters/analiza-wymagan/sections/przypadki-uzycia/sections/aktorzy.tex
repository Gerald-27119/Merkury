\subsection{Aktorzy}
\label{subsec:aktorzy}
\begin{description}[style=unboxed,leftmargin=0pt,labelindent=0pt]
    \item[\textbf{Użytkownik}] - Reprezentuje każdą osobę korzystającą z aplikacji.
    \item[\textbf{Usługa SMTP}] - usługa Simple Mail Transfer Protocol wykorzystywana do wysyłania wiadomości e-mail.
    \item[\textbf{Usługa OAuth}] - usługa uwierzytelniania i autoryzacji użytkowników z wykorzystaniem zewnętrznych dostawców tożsamości.
    \item[\textbf{Usługa do przechowywania plików w chmurze}] - magazyn plików w chmurze służący do przechowywania załączników i multimediów użytkowników.
    \item[\textbf{Usługa do wyświetlania mapy}] - zewnętrzne API dostarczające kafelki map, nawigację oraz dane geolokalizacyjne.
    \item[\textbf{Usługa danych pogodowych}] - usługa udostępniająca bieżące warunki pogodowe oraz prognozy dla wybranych lokalizacji.
    \item[\textbf{Usługa do GIF'ów}] - serwis umożliwiający wyszukiwanie i osadzanie animowanych obrazów GIF w aplikacji.
    \item[\textbf{Usługa do określania strefy czasowej}] - usługa ustalająca strefę czasową spota na podstawie jego współrzędnych geograficznych.
\end{description}
