\subsection{Endpointy systemu:}
\label{subsec:endpointy-systemu}


\subsubsection*{\lstinline[language=http]{GET} \texttt{/user-dashboard/profile}}

\noindent\textbf{Opis:} Zwraca profil aktualnie zalogowanego użytkownika.\\[2pt]
\textbf{Metoda:} \lstinline[language=http]{GET /user-dashboard/profile}\\[2pt]
\textbf{Zwraca (200 OK):} \texttt{application/json} — obiekt \texttt{UserProfileDto}.\\[2pt]
\textbf{Błąd (404 Not Found):} \texttt{text/plain} — \newline komunikat z wyjątku \texttt{UserNotFoundByUsernameException}.\\[6pt]

\noindent\textbf{Przykładowa odpowiedź (200 OK):}
\begin{lstlisting}[language=json]
{
  "username": "john_doe",
  "profilePhoto": "https://cdn.example.com/profiles/john_doe.jpg",
  "followersCount": 125,
  "followedCount": 87,
  "friendsCount": 32,
  "photosCount": 58,
  "mostPopularPhotos": [
    {
      "src": "https://cdn.example.com/photos/123.jpg",
      "heartsCount": 240,
      "viewsCount": 3400,
      "title": "Sunset at the beach",
      "id": 123
    }
  ]
}
\end{lstlisting}
