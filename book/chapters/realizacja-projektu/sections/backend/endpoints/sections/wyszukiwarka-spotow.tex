\subsubsection{Wyszukiwarka spotów}
\label{subsubsec:wyszukiwarka-spotow}

\cardendpoint
{EP12}
{/public/spot/search/home-page}
{Wyszukaj spoty na stronie głównej na podstawie lokalizacji}
{Brak}
{%
    \begin{tabitemize}
        \item \textbf{country}: String (opcjonalnie, kraj)
        \item \textbf{region}: String (opcjonalnie, region)
        \item \textbf{city}: String (opcjonalnie, miasto)
        \item \textbf{userLongitude}: Double (opcjonalnie, długość geograficzna użytkownika)
        \item \textbf{userLatitude}: Double (opcjonalnie, szerokość geograficzna użytkownika)
        \item \textbf{page}: Integer (numer strony, domyślnie 0)
        \item \textbf{size}: Integer (rozmiar strony, domyślnie 20)
    \end{tabitemize}
}
{200 OK}
{%
    \begin{tabitemize}
        \item \textbf{items}: List<HomePageSpotDto> (lista znalezionych spotów), każdy element:
        \begin{itemize}
            \item \textbf{id}: Long (identyfikator spota)
            \item \textbf{name}: String (nazwa spota)
            \item \textbf{rating}: Double (średnia ocena spota)
            \item \textbf{ratingCount}: Integer (liczba ocen spota)
            \item \textbf{firstPhoto}: String (URL pierwszego zdjęcia spota)
            \item \textbf{tags}: Set<SpotTagDto> (zestaw tagów przypisanych do spota)
            \item \textbf{centerPoint}: BorderPoint (punkt centralny obszaru spota)
            \item \textbf{city}: String (miasto, w którym znajduje się spot)
            \item \textbf{distanceToUser}: Double (odległość od lokalizacji użytkownika, jeśli dostępna)
        \end{itemize}
        \item \textbf{hasNext}: boolean (czy istnieje kolejna strona wyników)
    \end{tabitemize}
}
